pipeline {
    agent {
        node {
            label 'Win'
        }
    }
    
    options {
        timestamps()
        buildDiscarder(logRotator(artifactNumToKeepStr: '2', numToKeepStr: '2'))
        timeout(time: 1, unit: 'HOURS')
        disableConcurrentBuilds()
    }
    
    stages {
        stage('Start') {
            steps {
                echo "Process Start"
            }
        }
        stage('Docker build') {
            steps {
                bat 'docker build -t kadintisai/nginx-snake:V1 -f /root/Nginx_dir/Dockerfile .'
            }
        }
        stage('Docker Push') {
            steps {
                bat 'docker push kadintisai/nginx-snake:V1'
            }
        }
        stage('Docker Run') {
            steps {
                bat 'docker run -p 8080:80 --name custom-snake-1 -d kadintisai/nginx-snake:V1'
            }
        }
    }
    post {
        success {
            echo 'The pipeline succeeded...!!'
        }
        failure {
            echo 'The pipeline failed....!!'
        }
    }
}
